CADERNO MYSQL

COMANDOS BÁSICOS:

create database cadastro
default character set utf8
default collate utf8_general_ci;

drop database meubanco;

CRIANDO UMA TABELA:

use cadastro; create table pessoas(
id int not null auto_increment,
nome varchar(30) not null,
nascimento date,
sexo enum('M','F'),
peso decimal(5,2),
altura decimal(3,2),
nacionalidade varchar(20) default 'Brasil',
primary key (id)
)

INSERINDO DADOS NA TABELA:

insert into pessoas
(nome,nascimento,sexo,peso,altura,nacionalidade)
values
('Creuza','1920-01-02','F','50.2','1.65',DEFAULT);

//se a ordem for a msm da tabela é possivel ocultar os parametros

select * from pessoas;

MAIS MANIPULAÇOES DE TABELA:

Adicionar coluna no final
alter table pessoas
add column profissao varchar(10);

Adicionar depois de alguma coluna
alter table pessoas
add column profissao varchar(10) after nome;

Adicionar na primeira posição
alter table pessoas
add column codigo int first;

Modificar tabela
alter table pessoas
modify column profissao varchar(20) not null default '';

Mudar nome tabela
alter table pessoas
change column profissao prof varchar(20) not null;
ou
alter table pessoas
rename to gafanhotos;

MANIPULANDO LINHAS

Mudar um elemento de uma linha

UPDATE cursos
set nome = 'HTML5'
where idcurso = '1';

Mudar mais de 1 elemento de uma linha

UPDATE cursos
